        public static boolean CabinetNumberVerification(String strCode) {
        boolean result = true;
        try {
            if (strCode.length() != 11) {
                return false;
            }
            /*特殊处理逻辑 将入参柜号前四位转为大写，再去校验*/
            String firstFour = strCode.substring(0, 4);
            String fourAfter = strCode.substring(4);
            strCode = firstFour.toUpperCase() + fourAfter;
            /*特殊处理逻辑 将入参柜号前四位转为大写，再去校验*/
            char[] codeChars = strCode.toCharArray();
            String charCode = "0123456789A?BCDEFGHIJK?LMNOPQRSTU?VWXYZ";
            int num = 0;
            for (int i = 0; i < 10; i++) {
                int idx = charCode.indexOf(codeChars[i]);
                if (idx == -1 || charCode.charAt(idx)=='?') {
                    return false;
                }
                idx =  (int) (idx * Math.pow(2, i));
                num += idx;
            }
            num = (num % 11) % 10;
            result = Integer.parseInt(String.valueOf(codeChars[10])) == num;
        }catch (Exception e){
            result = false;
        }
        return  result;
    }

    Function power2Func(power As Integer)
   If power = 0 Then
    power2Func = 1
   Else
   power2Func = 1
    For cnum = 1 To power Step 1
     power2Func = power2Func * 2
    Next cnum

   End If

End Function

Sub validCabinetNo()


Dim initNo As Integer
initNo = 1
lineNo = Range("B" & Rows.Count).End(xlUp).Row - initNo

 If lineNo = 0 Then
    MsgBox ("请输入需要校验的柜号!")
    Exit Sub
 End If
 
'MsgBox (power2Func(5))
    
 '循环处理
 For num = 1 To lineNo Step 1
 
    Cells(num + initNo, "A") = num
    curCell = Cells(num + initNo, "B")
    'Cells(num + initNo, "C") = IIf(curCell = vbNullString Or Len(curCell) <> 11, False, True)
    
    If curCell = vbNullString Or Len(curCell) <> 11 Then
        Cells(num + initNo, "C") = "不符合"
        Range("C" & (num + initNo)).Interior.Color = 255
        Range("B" & (num + initNo)).Interior.Color = 255
        GoTo continueNext
    End If
    
'将入参柜号前四位转为大写，再去校验
    firstFour = UCase(Mid(curCell, 1, 4))
    fourAfter = Mid(curCell, 5)
    newCell = firstFour + fourAfter
    
    charCode = "0123456789A?BCDEFGHIJK?LMNOPQRSTU?VWXYZ"
    
    selfCounter = 0
'处理逻辑
    snum = 0
    Dim cc As Integer
    For cc = 0 To 9 Step 1
    
        idx = InStr(1, charCode, Mid(newCell, cc + 1, 1), vbTextCompare) - 1
        
        
        
        If idx = -1 Or Mid(newCell, cc + 1, 1) = "?" Then
            Cells(num + initNo, "C") = "不符合"
            Exit For
        End If
        
        'MsgBox (power2Func(cc))
        
        idx = idx * power2Func(cc)
        snum = snum + idx
        
        selfCounter = selfCounter + 1
        If selfCounter = 100 Then
            MsgBox ("运行有误，请联系Dereklin！")
            Exit For
        End If
        
    Next cc
    
    snum = (snum Mod 11) Mod 10
    'Debug.Print snum
    'gg = Mid(newCell, 11, 1)
    'aa = Trim(Str(snum))

    Cells(num + initNo, "C") = IIf(StrComp(Mid(newCell, 11, 1), Trim(Str(snum)), 1) = 0, "合格", "不符合")
    
    If Range("C" & (num + initNo)).Value = "不符合" Then
        Range("C" & (num + initNo)).Interior.Color = 255
        Range("B" & (num + initNo)).Interior.Color = 255
    Else
        Range("C" & (num + initNo)).Interior.Color = 5287936
        Range("B" & (num + initNo)).Interior.Color = 5287936
    End If

continueNext:
Next num

MsgBox ("校验完成")

End Sub
